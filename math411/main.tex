\begin{definition} 
	$\mathbb{R}^n = \{(u_1,u_2,\hdots,u_n) : u_i \in \mathbb{R} \forall i\}$
\end{definition}

\begin{definition} 
	Let $u,v\in \mathbb{R}^n$. $\ip{u}{v} = u_1v_2 + u_2v_2 + \hdots + u_nv_n$
\end{definition}

\begin{theorem} 
	Let $u,v\in \mathbb{R}^n$. Then $|\langle u,b \rangle| \leq ||u|||v||$.
\end{theorem}

\begin{proof} 
	Minimize $||u-\lambda v ||^2$ wrt $\lambda$.
\end{proof}
